Metrik üretecek nodejs uygulamasıdır. 
global:
scrape_interval: 5s

scrape_configs:

- job_name: "sabit-dosya"
  metrics_path: '/metrics.html'
  static_configs:
  - targets: ["host.docker.internal:3000"]

  - job_name: "counter"
  metrics_path: '/counter'
  static_configs:
  - targets: ["host.docker.internal:3000"]

  - job_name: "gauge"
  metrics_path: '/gauge'
  static_configs:
  - targets: ["host.docker.internal:3000"]

  - job_name: "summary"
  metrics_path: '/summary'
  static_configs:
  - targets: ["host.docker.internal:3000"]
